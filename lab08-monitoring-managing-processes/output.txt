Task 2: Sending Signals
--------------------------------------------------

-bash-4.2$ sleep 300 &
[1] 1250

-bash-4.2$ pidof sleep
1250

-bash-4.2$ kill 1250
[1]+  Terminated              sleep 300


-bash-4.2$ sleep 300 &
[1] 1284

-bash-4.2$ kill -9 1284
[1]+  Killed                  sleep 300


-bash-4.2$ sleep 300 &
[1] 1310

-bash-4.2$ pkill sleep
[1]+  Terminated              sleep 300

--------------------------------------------------
Task 4: Adjusting Process Priority
--------------------------------------------------

-bash-4.2$ nice -n 10 sleep 300 &
[2] 1378

-bash-4.2$ ps -o pid,ni,cmd -p 1378
  PID  NI CMD
 1378  10 sleep 300

-bash-4.2$ pgrep sleep
1378

-bash-4.2$ sudo renice -n 15 -p 1378
1378 (process ID) old priority 10, new priority 15

-bash-4.2$ ps -o pid,ni,cmd -p 1378
  PID  NI CMD
 1378  15 sleep 300
